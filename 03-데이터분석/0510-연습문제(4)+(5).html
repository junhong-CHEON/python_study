from matplotlib import pyplot
pyplot.rcParams['font.family'] = 'Malgun Gothic'
pyplot.rcParams['font.size'] = 16
pyplot.rcParams['figure.figsize'] = (15, 10)

# [1학년, 2학년, 3학년, 4학년]
xlabels= []
for i in mdf.index:
    xlabels.append('%d학년' % i)

pyplot.figure()
# x좌표는 1,2,3,4   y좌표는 평균점수
pyplot.bar(mdf.index, mdf.values, label='평균점수')
pyplot.legend()
pyplot.xticks(mdf.index, xlabels)
pyplot.grid()

# mdf.values는 numpy 배열이므로
# mdf.values에 대한 index 값인 i는 0,1,2,3
for i, v in enumerate(mdf.values):
    s = "%0.1f점" % v
    
    # 그래프의 x좌표는 1,2,3,4이지만 i는 0,1,2,3이므로 i+1
    pyplot.text(i+1, v, s, color='red', fontsize=14, horizontalalignment='center', verticalalignment='bottom')

pyplot.show()
pyplot.close()
